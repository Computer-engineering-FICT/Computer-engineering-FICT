private int publicationsNumber; /** * */ public Assistant() { academicRank = RANK; } /** * @param surname * @param name */ public Assistant(String surname, String name) { super(surname, name); academicRank = RANK; } /** * @param surname * @param name * @param facultee * @param cathedra * @param position */ public Assistant(String surname, String name, String facultee, String cathedra, String position) { super(surname, name, facultee, cathedra, position); academicRank = RANK; } /** * @param surname * @param name * @param facultee * @param cathedra * @param position * @param degree */ public Assistant(String surname, String name, String facultee, String cathedra, String position, String degree) { super(surname, name, facultee, cathedra, position); academicRank = RANK; academicDegree = degree; } /* (non-Javadoc) * @see Academic#canPromotePosition() */ @Override public boolean canPromotePosition() { return (((academicDegree.compareTo("Candidate of science") == 0) || (academicDegree.compareTo("Doctor") == 0)) && (publicationsNumber > 1)); } }
